from pyomo.environ import *

def solve():
    '''
    Knapsack problem with 25 items and capacity 625.
    
    Returns
    -------
    float
        Optimal objective value (total value of selected items).
        
    Raises
    ------
    Exception
        If solver fails to find optimal solution.
    '''
    NUM_ITEMS = 25
    model = ConcreteModel()
    
    # Item data
    values = [327, 314, 231, 184, 310, 312, 287, 273, 280, 329,
              259, 309, 249, 189, 251, 226, 201, 300, 257, 257,
              279, 223, 152, 300, 209]
    weights = [29, 48, 30, 42, 45, 39, 36, 46, 33, 55, 38, 51,
               38, 44, 43, 35, 31, 43, 33, 41, 35, 32, 39, 30, 26]
    capacity = 625
    
    # Validate inputs
    if len(values) != NUM_ITEMS or len(weights) != NUM_ITEMS:
        raise ValueError("Values and weights must have exactly 25 items")
    
    if capacity < 0:
        raise ValueError("Capacity must be non-negative")
    
    # Create variables
    model.x = Var(range(NUM_ITEMS), within=Binary)
    
    # Objective: maximize total value
    model.obj = Objective(
        expr=sum(values[i] * model.x[i] for i in range(NUM_ITEMS)),
        sense=maximize
    )
    
    # Constraint: weight limit
    model.weight_con = Constraint(
        expr=sum(weights[i] * model.x[i] for i in range(NUM_ITEMS)) <= capacity
    )
    
    # Solve
    solver = SolverFactory('glpk')
    results = solver.solve(model)
    
    # Check solution status
    if results.solver.termination_condition != TerminationCondition.optimal:
        raise Exception(f"Solver failed to find optimal solution. "
                       f"Termination condition: {results.solver.termination_condition}")
    
    optimal_value = value(model.obj)
    
    # Optional: Clean up
    del model
    
    return optimal_value
```
