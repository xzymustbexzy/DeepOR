    '''

        (3,0):42,(3,1):30,(3,2):34,(3,4):35,(3,5):28,(3,6):28,

        (6,0):18,(6,1):50,(6,2):13,(6,3):38,(6,4):12,(6,5):17
    }
    model.x = Var(cities, cities, within=NonNegativeReals)
    model.obj = Objective(
        expr=sum(costs[i,j] * model.x[i,j] for i in cities for j in cities
                 if i != j),
        sense=minimize
    )
    def flow_balance_rule(model, k):
        inflow = sum(model.x[i,k] for i in cities if i != k)
        outflow = sum(model.x[k,j] for j in cities if j != k)
        return inflow - outflow == net_demand[k]
    model.flow_balance = Constraint(cities, rule=flow_balance_rule)
    def capacity_rule(model, i, j):
        if i == j:
            return Constraint.Skip
        return model.x[i,j] <= capacities[i,j]
    model.capacity = Constraint(cities, cities, rule=capacity_rule)
    solver = SolverFactory('glpk')
    solver.solve(model)
    return value(model.obj)
